        // Proxy Click Function
        window.gta_setLanguage = function(langCode, containerElement) {
            const domain = window.location.hostname;

            // 1. If user selects English (Reset), aggressively wipe the translation cookies and reload
            if (langCode === '' || langCode === 'en') {
                document.cookie = 'googtrans=; Path=/; Expires=Thu, 01 Jan 1970 00:00:01 GMT;';
                document.cookie = 'googtrans=; Path=/; domain=.' + domain + '; Expires=Thu, 01 Jan 1970 00:00:01 GMT;';
                document.cookie = 'googtrans=; Path=/; domain=' + domain + '; Expires=Thu, 01 Jan 1970 00:00:01 GMT;';
                window.location.reload();
                return; // Stop execution here to ensure the reload happens
            }

            // 2. Otherwise, handle translation into a new language
            const select = document.querySelector('.goog-te-combo');

            if (select) {
                select.value = langCode;
                select.dispatchEvent(new Event('change', { bubbles: true, cancelable: true }));
            } else {
                // FALLBACK: Forcefully translate via cookies and reload
                document.cookie = "googtrans=/en/" + langCode + "; path=/; domain=." + domain;
                document.cookie = "googtrans=/en/" + langCode + "; path=/;";
                window.location.reload();
            }

            if (containerElement) {
                containerElement.classList.remove('is-open');
            }
        };

                            }
