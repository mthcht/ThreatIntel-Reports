				for (
					var i = 0;
					i < form.querySelectorAll('input[type=hidden]').length;
					i++
				) {
					var e = form.querySelectorAll('input[type=hidden]')[i];
					if (e.name !== 'limit') {
						formParams.push(
							encodeURIComponent(e.name) + '=' + encodeURIComponent(e.value)

				var queryString = formParams.join('&');
				return queryString;

		var debounce = function (func, wait, immediate) {
			var timeout;
			return function () {
				var context = this,
					args = arguments;
				var later = function () {
					timeout = null;
					if (!immediate) {
						func.apply(context, args);
					}
				};
				var callNow = immediate && !timeout;
				clearTimeout(timeout);
				timeout = setTimeout(later, wait || 200);
				if (callNow) {
					func.apply(context, args);
				}
			};
		},
			emptySearchResults = function () {
				searchResults.innerHTML = '';
				searchField.focus();
				searchForm.classList.remove('hs-search-field--open');
			},
			fillSearchResults = function (response) {
				var items = [];
				items.push(

				);
				response.results.forEach(function (val, index) {
					items.push(

						"'><a href='" +
						val.url +
						"'>" +
						val.title +
						'</a></li>'
					);
				});

				emptySearchResults();
				searchResults.innerHTML = items.join('');
				searchForm.classList.add('hs-search-field--open');
			},
			getSearchResults = function () {
				var request = new XMLHttpRequest();
				var requestUrl =
					'/_hcms/search?&term=' +

					'&limit=' +
					encodeURIComponent(TYPEAHEAD_LIMIT) +

<p>Play ransomware (also known as PlayCrypt) is a new ransomware operation that launched in June 2022. The operation has amassed a steady stream of victims across the world. Play has recently been in the news for attacking Argentina’s Judiciary of Cordoba and the German hotel chain “H-Hotels”. Play’s attacks focus on organizations in the Latin American region – Brazil being their primary target. They have also been observed deploying attacks on India, Hungary, Spain, and the Netherlands.</p>
<p>Play is known for their big game hunting tactics, such as using Cobalt Strike for post-compromise and SystemBC RAT for persistence. They have recently started exploiting the ProxyNotShell vulnerabilities in Microsoft Exchange. The group also has similar tactics and techniques to the ransomware groups Hive and Nokoyawa, leading researchers to believe Play is operated by the same people. Let’s take a look at Play ransomware, their tactics and techniques, as well as how organizations can protect themselves from this kind of threat actor.</p>

<h2 style="font-weight: normal;">play ransomware</h2>
<p>Play was discovered in June 2022 after several victims of their ransomware attacks appeared in Bleeping Computer forums. Play’s ransomware name stems from its behavior, as the extension <strong>.play</strong> is added after file encryption. The ransomware note left behind also contains the single word <strong>PLAY</strong>, as well as the group’s contact email address.</p>
<p>Not long after victims started reporting attacks in Bleeping Computer forums, Play targeted Argentina’s Judiciary of Cordoba. The attack took place in August 2022 and the Judiciary had to shut down their IT systems, as well as their online portal as a result. The Judiciary was forced to use pen and paper for submitting official documents. The Argentinian publication, <strong><em>Clarin,</em></strong> reported that the attack was the “worst attack on public institutions in history.”</p>

<h2>tactics &amp; techniques</h2>
<p>&nbsp;</p>
<h3 style="font-size: 28px; font-weight: normal;"><span style="color: #be3a34;">PROXYNOTSHELL</span></h3>
<p>Towards the end of December 2022, Play was observed using a method to exploit two <a href="/resources/threat-reports/microsoft-patches-zero-days" rel="noopener" target="_blank">ProxyNotShell</a> vulnerabilities in Microsoft exchange to gain initial access in environments. The ProxyNotShell flaws are:</p>
<ul>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span><strong>CVE-2022-41040</strong> – Microsoft Exchange Server Elevation of Privilege Vulnerability</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span><strong>CVE-2022-41082</strong> – Microsoft Exchange Server Remote Code Execution Vulnerability</li>
</ul>
<p>The flaws impact Exchange Server 2013, 2016, and 2019. If successful, an attacker could exploit ProxyNotShell to elevate privileges to run PowerShell in the context of the system, gaining arbitrary or remote code execution on vulnerable servers. Although the vulnerabilities were patched by Microsoft in November 2022, Play was able to bypass URL rewrite mitigations for the Autodiscover endpoint implemented by Microsoft.</p>
<p>Next, Play was able to leverage legitimate Plink and AnyDesk executables to maintain access. They also performed anti-forensics techniques on the server, attempting to hide their activity. The security researchers at CrowdStrike investigated several Play ransomware intrusions where they suspected the threat actors used ProxyNotShell as their entry vector. However, after analyzing and after reviewing their logs, they determined that there was no evidence of exploitation of CVE-2022-41040 for initial access but “corresponding requests were made directly through the Outlook Web Application (OWA) endpoint - signifying a previously undisclosed exploit method for Exchange.”</p>
<p>According to Security Affairs, a security researcher from Huntress Labs discovered Play’s tool set and shared it through MegaUpload. The tools include a Python script “<strong>poc.py</strong>” (which led CrowdStrike to replicate the logs generated in recent Play ransomware attacks).</p>
<p>&nbsp;</p>
<h3 style="font-size: 28px; font-weight: normal;"><span style="color: #be3a34;">CONNECTIONS TO OTHER GROUPS</span></h3>
<p>As previously stated, Play’s name originated from the <strong>.play</strong> extension that is added after encrypting files. The threat actors have similar behavior and tactics to <a href="https://www.avertium.com/resources/threat-reports/hive-ransomware-attacks-analysis">HIVE</a> and Nokoyawa ransomware. The threat actor was observed using similar file names and file paths of their respective tools and payloads. Trend Micro also found evidence hinting that the operators behind Nokoyawa are associated with the operators behind HIVE, as there are many similarities between their attack chains.</p>
<p>One of the ways that Play is set apart from <a href="/resources/threat-reports/an-update-on-hive-ransomware" rel="noopener" target="_blank">HIVE</a> and Nokoyawa is through their use of AdFind – a command-line query tool which collects information from Active Directory (AD). HIVE has been observed using tools such as the TrojanSpy.DATASPY trojan to gather victim information. Play, HIVE, and Nokoyawa may not share all the same indicators, but their shared tactics and tools indicate that they are affiliated in some way.</p>
<p>Trend Micro also found evidence of a possible connection between Play and <a href="/resources/threat-reports/an-in-depth-look-at-quantum-ransomware" rel="noopener" target="_blank">Quantum</a> ransomware. Quantum is known for being a splinter group of Conti. The ransomware operation was taken over by Conti Team Two in April 2022, but the group kept Quantum’s original name. Quantum is also known for employing their version of BazarCall, called Jormungandr, and hiring people who specialize in OSINT, spamming, design, and call center operations.</p>
<p>Trend Micro’s researchers stated that the Cobalt Strike beacons used in Play’s attacks, have the same watermark that was dropped by the <a href="/resources/threat-reports/an-in-depth-look-at-emotet-botnet" rel="noopener" target="_blank">Emotet</a> and SVCReady botnets – two botnets that have been observed in Quantum ransomware attacks.</p>
<p>&nbsp;</p>
<p style="text-align: center;"><span style="font-weight: bold; color: #ffffff; background-color: #be3a34;">Image 4:</span> <span style="color: #be3a34;">Play Infection Train</span></p>
<p style="text-align: center;"><a href="http://trendmicro.com" rel="noopener" target="_blank"><img src="https://www.avertium.com/hs-fs/hubfs/tr4-Jan-04-2023-03-21-21-6720-PM.png?width=507&amp;height=415&amp;name=tr4-Jan-04-2023-03-21-21-6720-PM.png" alt="Play Infection Chain" width="507" height="415" loading="lazy" style="height: auto; max-width: 100%; width: 507px;" srcset="https://www.avertium.com/hs-fs/hubfs/tr4-Jan-04-2023-03-21-21-6720-PM.png?width=254&amp;height=208&amp;name=tr4-Jan-04-2023-03-21-21-6720-PM.png 254w, https://www.avertium.com/hs-fs/hubfs/tr4-Jan-04-2023-03-21-21-6720-PM.png?width=507&amp;height=415&amp;name=tr4-Jan-04-2023-03-21-21-6720-PM.png 507w, https://www.avertium.com/hs-fs/hubfs/tr4-Jan-04-2023-03-21-21-6720-PM.png?width=761&amp;height=623&amp;name=tr4-Jan-04-2023-03-21-21-6720-PM.png 761w, https://www.avertium.com/hs-fs/hubfs/tr4-Jan-04-2023-03-21-21-6720-PM.png?width=1014&amp;height=830&amp;name=tr4-Jan-04-2023-03-21-21-6720-PM.png 1014w, https://www.avertium.com/hs-fs/hubfs/tr4-Jan-04-2023-03-21-21-6720-PM.png?width=1268&amp;height=1038&amp;name=tr4-Jan-04-2023-03-21-21-6720-PM.png 1268w, https://www.avertium.com/hs-fs/hubfs/tr4-Jan-04-2023-03-21-21-6720-PM.png?width=1521&amp;height=1245&amp;name=tr4-Jan-04-2023-03-21-21-6720-PM.png 1521w" sizes="(max-width: 507px) 100vw, 507px"></a></p>
<p style="text-align: center; font-size: 14px;"><em>Source: Trend Micro</em></p>
<p style="text-align: center; font-size: 14px;">&nbsp;</p>
<p>As you can see above, Play’s infection chain includes using compromised valid accounts or unpatched Fortinet SSL VPN vulnerabilities to gain access to organization’s networks. They use living-off-the-land binaries (LOLBins) during their attacks (i.e., using WinSCP for data exfiltration and Task Manger for Local Security Authority Server Service for dumping and credential cracking). Like other ransomware gangs, Play uses the double extortion model against their victims. The threat actor exfiltrates data before deploying their ransomware and archives the victim’s files using WinRAR before uploading them to sharing sites.</p>
<p>Regarding execution, Trend Micro observed Play using scheduled tasks and PsExec during execution. The ransomware gang also creates a GPO, which can control user and machine settings in the AD. Play uses batch files to execute PsExec, which executes processes on other systems and allows the spread of the ransomware. This also assists Play in its reconnaissance activities.</p>
<p>After gaining access to accounts, Play uses the accounts as a persistence mechanism. If Remote Desktop Protocol (RDP) is disabled on their victim’s system, Play will enable it by executing “<strong>netsh</strong>” commands – establishing inbound connects within the victim’s system. Next Mimikatz is used to extract high privilege credentials from memory. To avoid detection, GMER, IOBit, Process Hacker, and PowerTool is used to disable antivirus tools and monitoring solutions. Wevtutil or a batch script is used to remove indicators that show Play was present in a system, while Windows Defender is disabled through PowerShell or command script.</p>
<p style="font-weight: bold;">Play uses the following tools to move laterally across a victim’s system:</p>
<ul>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Cobalt Strike SMB beacon</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>SystemBC</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Empire</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Mimikatz</li>
</ul>
<p>To avoid triggering network data transfer, Play splits victim data into chunks before exfiltration. To compress the files in <strong>.RAR</strong> format, they use WinRaR.</p>
<h2>&nbsp;</h2>
<h2>&nbsp;</h2>
<h2>what organizations can do</h2>
<p style="font-weight: bold;"><span style="color: #be3a34;">here are some best practices that can help protect organizations and end users from ransomware operators such as Play, those best practices include:</span></p>
<ul>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>If running Exchange servers, ensure they are fully patched and running on Windows Server 2019. Or move to M365 and a fully hosted environment.</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Block or restrict the use of <strong>psexec</strong> within your environment.</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Minimize the use of administrator accounts (at both the local and domain level). All domain administrators should have standard accounts for day-to-day use and additional administrator accounts that are only used when required.</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Use Data Loss Prevention (DLP) software to detect and block aggregation and exfiltration of sensitive data.</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Where possible disable the use of Remote Desktop Protocol via GPO and do not allow users to install other remote access software.</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Back up all data and test those backups regularly. Even if you become the victim of a ransomware attack, your business operations won’t be severely interrupted, and your data will be retrievable.</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Install updates regularly – maintain patch management protocols and keep your operating systems and applications up to date. This action will deter threat actors from exploiting software vulnerabilities.</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Using phishing resistant MFA on all accounts, especially VPNs, webmail, and accounts with access to critical systems will prevent attackers from performing lateral movement inside a network.</li>
<li style="margin-top: 0in; margin-right: 0in; margin-bottom: 8pt;"><span style="color: #c00000;"> </span>Deploy and monitor Endpoint Detection and Response or antivirus software and ensure that your analysts are trained to determine at what point of the cyber attack the detection relates to.</li>
</ul>
<p>Remember, relying on outdated tools and point solutions will only compromise your network or system. Organizations should rely on modern technology to detect complex attacks.</p>

<h2 style="padding-left: 0in;">avertium's recommendations <strong>&nbsp;</strong><strong>&nbsp;</strong><strong>&nbsp;</strong></h2>
<p><span style="color: #be3a34;"><strong>The FBI and CISA urge all organizations to apply the following recommendations to prepare for, mitigate/prevent, and respond to ransomware incidents:</strong></span></p>
<ul>
<li>Review the security posture of third-party vendors and those interconnected with your organization. Ensure all connections between third-party vendors and outside software or hardware are monitored and reviewed for suspicious activity.<br><br></li>
<li>Implement a recovery plan&nbsp;to maintain and retain multiple copies of sensitive or proprietary data and servers in a physically separate, segmented, and secure location (i.e., hard drive, storage device, the cloud).<br><br></li>
<li>Require all accounts&nbsp;with password logins (e.g., service account, admin accounts, and domain admin accounts) to comply with&nbsp;<a href="https://pages.nist.gov/800-63-3/" rel="noopener" target="_blank">National Institute of Standards and Technology (NIST) standards</a>&nbsp;for developing and managing password policies.<br><br></li>
<li>Review domain controllers, servers, workstations, and active directories&nbsp;for new and/or unrecognized accounts.<br><br></li>
<li>Segment networks&nbsp;to prevent the spread of ransomware. Network segmentation can help prevent the spread of ransomware by controlling traffic flows between—and access to—various subnetworks and by restricting adversary lateral movement.<br><br></li>
<li>Consider adding an email banner to emails&nbsp;received from outside your organization.<br><br></li>
<li>Disable command-line and scripting activities and permissions. Privilege escalation and lateral movement often depend on software utilities running from the command line. If threat actors are not able to run these tools, they will have difficulty escalating privileges and/or moving laterally.</li>
</ul>

				if (ancestor.matches(s)) return ancestor;
				ancestor = ancestor.parentElement;
			} while (ancestor !== null);
			return null;
		};
	}


	/*
	Vanilla Debounce -- Debounce for all the things!!!!
	http://davidwalsh.name/function-debounce
		1. params: function to be called, wait time, or if needed immediately
		2. Use this you monsters!
		i.e. window.addEventListener('scroll', debounce(func, wait));
	*/
	function debounce(func, wait, immediate) {
		var timeout;
		return function () {
			var context = this, args = arguments;
			var later = function () {
				timeout = null;
				if (!immediate) func.apply(context, args);
			};
			var callNow = immediate && !timeout;
			clearTimeout(timeout);
			timeout = setTimeout(later, wait);
			if (callNow) func.apply(context, args);
		};
	};
