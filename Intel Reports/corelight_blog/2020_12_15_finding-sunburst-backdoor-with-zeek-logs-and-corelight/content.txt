<script>
  document.addEventListener('DOMContentLoaded', function() {
    const postBody = document.querySelector('.atlas-blog-post__body');
    const tocList = document.getElementById('atlas-blog-post__toc-list');
    if (!postBody || !tocList) return;
    
    const headings = postBody.querySelectorAll('h2');
    const tocLinks = [];
    
    headings.forEach((heading) => {
      // Create ID from heading text
      const id = heading.textContent
        .toLowerCase()
        .trim()
        .replace(/[^\w\s-]/g, '')
        .replace(/\s+/g, '-')
        .replace(/-+/g, '-');
      
      // Create and insert anchor element before heading
      const anchor = document.createElement('a');
      anchor.setAttribute('data-hs-anchor', 'true');
      anchor.id = id;
      anchor.style.display = 'block';
      anchor.style.position = 'relative';
      anchor.style.top = '-125px';
      anchor.style.visibility = 'hidden';
      heading.parentNode.insertBefore(anchor, heading);
      
      // Create TOC link
      const li = document.createElement('li');
      li.className = 'atlas-blog-post__toc-item';
      
      const a = document.createElement('a');
      a.className = 'atlas-blog-post__toc-link';
      a.href = '#' + id;
      a.textContent = heading.textContent;
      
      li.appendChild(a);
      tocList.appendChild(li);
      tocLinks.push({ link: a, heading: heading });
    });
    
    // Update active class on scroll
    function updateActiveLink() {
      const scrollPosition = window.scrollY + 150;
      let activeLink = null;
      
      // Find the last heading that's above the scroll position
      for (let i = tocLinks.length - 1; i >= 0; i--) {
        const headingTop = tocLinks[i].heading.offsetTop;
        if (scrollPosition >= headingTop) {
          activeLink = tocLinks[i].link;
          break;
        }
      }
      
      // Remove active class from all links
      tocLinks.forEach(({ link }) => link.classList.remove('active'));
      
      // Add active class to current section
      if (activeLink) {
        activeLink.classList.add('active');
      }
    }
    
    window.addEventListener('scroll', updateActiveLink);
    updateActiveLink(); // Initial check
  });
</script>

    function initializeCustomPagesAlertBarSwiper() {
        
        // Store HubSpot module variables in JavaScript variables
        var sliderDirectionSetting = "vertical";

        var sliderSpeed = parseInt("2000", 10) || 300;

            return;
        }

        // Check if there are slides
        var slides = document.querySelectorAll('.custom-pages-alert-banner .swiper-slide');
        
        // If no slides found, exit early
        if (!slides || slides.length === 0) {
            return;
        }

        // Determine slider direction
        var slider_direction = 'horizontal'; // Default

        if (sliderDirectionSetting === 'vertical') {
            slider_direction = 'vertical';
        }
        // For 'none' or 'horizontal', keep the default 'horizontal'

        // If direction is 'none', don't initialize swiper
        if (sliderDirectionSetting === 'none') {
            // Just show the first slide
            var firstSlide = slides[0];
            if (firstSlide && firstSlide.classList) {
                firstSlide.classList.remove('hide');
            }
            return;
        }

        // If there's only one slide and no autoplay, don't initialize swiper
        if (slides.length === 1 && sliderDuration === 0) {
            var firstSlide = slides[0];
            if (firstSlide && firstSlide.classList) {
                firstSlide.classList.remove('hide');
            }
            return;
        }


        // Verify all slides are valid DOM elements before initializing
        var validSlides = [];
        for (var i = 0; i < slides.length; i++) {
            if (slides[i] && slides[i].nodeType === 1) { // Node.ELEMENT_NODE
                validSlides.push(slides[i]);
            }
        }
        
        // If no valid slides, exit
        if (validSlides.length === 0) {
            return;

                    validChildren.push(finalCheck.children[j]);
                }
            }
            if (validChildren.length === 0) {
                return;
            }
            
            // Additional safety: verify slides array matches valid children
            if (validSlides.length !== validChildren.length) {
                validSlides = validChildren;

            // Don't initialize Swiper if we have invalid slides
            if (validSlides.length === 0) {
                return;
            }
            
            // Initialize Swiper based on duration setting

          <span id="hs_cos_wrapper_post_body" class="hs_cos_wrapper hs_cos_wrapper_meta_field hs_cos_wrapper_type_rich_text" style="" data-hs-cos-general-type="meta_field" data-hs-cos-type="rich_text"><p><strong>UPDATE 12-16-20: Corelight Resources</strong></p>
<!--more-->
<ul>
<li>WEBCAST RECORDING – Finding SolarWinds backdoors with Zeek, Suricata &amp; Corelight –&nbsp;<a rel="noreferrer noopener" href="https://go.corelight.com/finding-sunburst-solarwinds-zeek-suricata" target="_blank"><strong>watch here</strong></a></li>
<li>WEBCAST SLIDE DECK –&nbsp;<a rel="noreferrer noopener" href="https://go.corelight.com/corelight-presentation-find-solarwinds-sunburst-backdoors-zeek-suricata" target="_blank"><strong>download link</strong></a></li>
<li>IOCs SPREADSHEET –&nbsp;<a rel="noreferrer noopener" href="https://solarigate.training.corelight.io/" target="_blank"><strong>Corelight/Zeek Queries Table – to hunt for Sunburst IOCs</strong></a></li>
</ul>
<p><strong>——————————————–</strong></p>
<p>FireEye’s threat research team has <strong><a rel="noreferrer noopener" aria-label="discovered (opens in a new tab)" href="https://www.fireeye.com/blog/threat-research/2020/12/evasive-attacker-leverages-solarwinds-supply-chain-compromises-with-sunburst-backdoor.html" target="_blank">discovered</a></strong> a troubling new supply chain attack targeting SolarWind’s <a rel="noreferrer noopener" aria-label="Orion (opens in a new tab)" href="https://www.solarwinds.com/solutions/orion" target="_blank"><strong>Orion</strong></a> IT monitoring and management platform. The attack trojanizes Orion software updates to deliver malware called SUNBURST, which opens a stealthy backdoor for command-and-control and other malicious activity that blends in with Orion Improvement Program (OIP) protocol traffic.&nbsp;</p>
<p>Scott Runnels, a Mandiant researcher involved in the discovery, <strong><a href="https://twitter.com/srunnels/status/1338329916304199680?s=20" target="_blank" rel="noreferrer noopener" aria-label="revealed (opens in a new tab)">revealed</a></strong> that Zeek played a key role in FireEye’s investigation and discovery of this new threat:&nbsp;</p>
<div class="wp-block-image">
<figure class="aligncenter size-large is-resized"><a href="https://twitter.com/srunnels/status/1338329916304199680" target="_blank" rel="noreferrer noopener"><img loading="lazy" src="https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/Screen-Shot-2020-12-14-at-3_24_54-PM-1.png?width=598&amp;height=298&amp;name=Screen-Shot-2020-12-14-at-3_24_54-PM-1.png" alt="2020-12-14-at-3_24_54-PM-1" class="wp-image-1571" width="598" height="298" srcset="https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/Screen-Shot-2020-12-14-at-3_24_54-PM-1.png?width=299&amp;height=149&amp;name=Screen-Shot-2020-12-14-at-3_24_54-PM-1.png 299w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/Screen-Shot-2020-12-14-at-3_24_54-PM-1.png?width=598&amp;height=298&amp;name=Screen-Shot-2020-12-14-at-3_24_54-PM-1.png 598w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/Screen-Shot-2020-12-14-at-3_24_54-PM-1.png?width=897&amp;height=447&amp;name=Screen-Shot-2020-12-14-at-3_24_54-PM-1.png 897w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/Screen-Shot-2020-12-14-at-3_24_54-PM-1.png?width=1196&amp;height=596&amp;name=Screen-Shot-2020-12-14-at-3_24_54-PM-1.png 1196w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/Screen-Shot-2020-12-14-at-3_24_54-PM-1.png?width=1495&amp;height=745&amp;name=Screen-Shot-2020-12-14-at-3_24_54-PM-1.png 1495w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/Screen-Shot-2020-12-14-at-3_24_54-PM-1.png?width=1794&amp;height=894&amp;name=Screen-Shot-2020-12-14-at-3_24_54-PM-1.png 1794w" sizes="(max-width: 598px) 100vw, 598px"></a></figure>
</div>
<p>Given the widespread use of the Orion software we want to provide the community and our customers with some preliminary guidance on how to use Zeek and related tools to manually find and automatically detect this novel threat in their environment.&nbsp;&nbsp;</p>
<p>We will host a webinar this <a href="https://go.corelight.com/finding-sunburst-solarwinds-zeek-suricata" target="_blank" rel="noreferrer noopener" aria-label="Wednesday, Dec. 16 t (opens in a new tab)"><strong>Wednesday, Dec. 16</strong> t</a>o deep dive on these methods and tools, which include:&nbsp;</p>
<ul>
<li><strong><a href="https://go.corelight.com/zeek-logs-v3.0" target="_blank" rel="noreferrer noopener" aria-label="Zeek log queries (opens in a new tab)">Zeek log queries</a></strong>: Network IOCs for this attack span a range of protocols parsed by Zeek including&nbsp; DNS, HTTP, and X509 certificates. Targeted queries in your SIEM against Zeek logs can reveal potential evidence of compromise related to this attack, for example:&nbsp;</li>
</ul>
<div class="wp-block-image">
<figure class="aligncenter size-large"><img loading="lazy" width="1829" height="295" src="https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/image-2-1.png?width=1829&amp;height=295&amp;name=image-2-1.png" alt="image-2-1.png" class="wp-image-1574" srcset="https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/image-2-1.png?width=915&amp;height=148&amp;name=image-2-1.png 915w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/image-2-1.png?width=1829&amp;height=295&amp;name=image-2-1.png 1829w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/image-2-1.png?width=2744&amp;height=443&amp;name=image-2-1.png 2744w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/image-2-1.png?width=3658&amp;height=590&amp;name=image-2-1.png 3658w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/image-2-1.png?width=4573&amp;height=738&amp;name=image-2-1.png 4573w, https://corelight.com/hs-fs/hubfs/Imported_Blog_Media/image-2-1.png?width=5487&amp;height=885&amp;name=image-2-1.png 5487w" sizes="(max-width: 1829px) 100vw, 1829px"></figure>
</div>
<ul>
<li><strong><a href="https://socprime.com/blog/sunburst-backdoor-detection-solarwinds-supply-chain-attack-on-fireeye-and-us-agencies/" target="_blank" rel="noreferrer noopener" aria-label="Sigma rules/queries (opens in a new tab)">Sigma rules/queries</a></strong>: Community-developed Sigma rules to detect SUNBURST are available in SOC Prime’s Threat Detection Marketplace, which you can access <a href="https://socprime.com/blog/sunburst-backdoor-detection-solarwinds-supply-chain-attack-on-fireeye-and-us-agencies/" target="_blank" rel="noreferrer noopener" aria-label="here (opens in a new tab)"><strong>here</strong></a>.&nbsp;Corelight customers with supported SIEM platforms (Splunk, Elastic, Humio, QRadar, ArcSight, Chronicle, et al.) can copy/paste the queries and/or detections directly into their SIEM environment.&nbsp;</li>
</ul>
<ul>
<li><a href="https://rules.emergingthreats.net/open/suricata-5.0/" target="_blank" rel="noreferrer noopener" aria-label="Suricata Rules in ET Open Ruleset (opens in a new tab)"><strong>Suricata Rules in ET Open Ruleset</strong></a>: Proofpoint Emerging Threats has added detections as Suricata rules in their latest ET Open Ruleset release, which you can download <a href="https://rules.emergingthreats.net/open/suricata-5.0/" target="_blank" rel="noreferrer noopener" aria-label="here (opens in a new tab)"><strong>here</strong></a>.&nbsp;Corelight customers with AP 200, AP 1001, and/or AP 3000 Sensors and a Suricata subscription can download and run these rules on their sensors.</li>
</ul>
<p>Again, we will host a webinar on Wednesday, Dec. 16 at 7a PST / 10a EST / 3p GMT to deep dive on these methods and tools.&nbsp;</p>
