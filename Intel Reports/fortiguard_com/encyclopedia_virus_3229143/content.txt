  <script>
    (function(){
      function setDomainTitle(){
        try{
          var titleEl = document.getElementById('domain-title');
          if(!titleEl) return;
          titleEl.textContent = window.location.host || window.location.hostname || 'unknown host';
        }catch(e){ /* ignore */ }
      }

      function setSecurityReview(){
        try{
          var reviewEl = document.getElementById('security-review');
          if(!reviewEl) return;
          var domain = window.location.host || window.location.hostname || 'this site';
          reviewEl.textContent = domain + ' needs to review the security of your connection before proceeding.';
        }catch(e){ /* ignore */ }
      }

      function setScreenId(){
        try{
          var el = document.getElementById('screen-id');
          if(!el) return;
          var id = null;
          if (window.crypto && typeof crypto.randomUUID === 'function'){
            id = crypto.randomUUID();
          } else if (window.crypto && crypto.getRandomValues){
            var bytes = crypto.getRandomValues(new Uint8Array(8));
            id = Array.from(bytes).map(function(b){ return ('0'+b.toString(16)).slice(-2); }).join('').toUpperCase();
            id = 'SC-' + id.slice(0,12);
          } else {
            id = 'SC-' + Math.random().toString(36).slice(2,12).toUpperCase();
          }
          el.textContent = 'Screen ID: ' + id;
        }catch(e){ /* ignore */ }
      }

      function ready(){
        setDomainTitle();
        setSecurityReview();
        //setScreenId();
      }

      if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', ready);
      else ready();
    })();

    // When the altcha widget reports verified, ensure the token is included in the form and submit.
    (function(){
      function submitViaFetchFallback(form){
        try{
          var url = form.action || window.location.href;
          fetch(url, {method:'POST', credentials:'same-origin', headers:{'Content-Type':'application/x-www-form-urlencoded'}, body: ''})
            .then(function(res){
              if(res.redirected) window.location.href = res.url;
              else window.location.reload();
            })
            .catch(function(e){ console.error('fetch fallback failed', e); });
        }catch(e){ console.error('fetch fallback threw', e); }
      }

      function ensureAltchaField(token){
        try{
          var f = document.getElementById('security-form');
          if(!f) return;
          var existing = f.querySelector('input[name="altcha"]');
          if(existing){ existing.value = token || existing.value; return; }
          var inp = document.createElement('input');
          inp.type = 'hidden';
          inp.name = 'altcha';
          inp.value = token || '';
          f.appendChild(inp);
        }catch(e){ console.error('ensureAltchaField error', e); }
      }

      function readTokenFromWidget(alt){
        if(!alt) return null;
        var selectors = [
          'input[type="hidden"][name="altcha"]',
          'input[name="altcha"]',
          'input[type="hidden"][name*="altcha"]',
          'input[data-altcha]',
          'input[name*="altcha"]'
        ];

        function isLikelyToken(val){
          if(!val || typeof val !== 'string') return false;
          val = val.trim();
          if(val.length < 30) return false;
          // base64-ish check
          if(/^[A-Za-z0-9_\-]+=*$/.test(val)) return true;
          // or contains JSON-like fields
          if(val.indexOf('"')!==-1 || val.indexOf('{')!==-1) return true;
          return false;
        }

        function findIn(root){
          if(!root || !root.querySelector) return null;
          for(var i=0;i<selectors.length;i++){
            try{
              var el = root.querySelector(selectors[i]);
              if(!el) continue;
              // skip checkbox inputs
              var t = (el.getAttribute && el.getAttribute('type')) || el.type || '';
              if(t && t.toLowerCase() === 'checkbox') continue;
              // prefer elements whose value looks like a token
              var v = el.value || el.getAttribute('value') || el.getAttribute('data-value') || '';
              if(isLikelyToken(v)) return el;
              // otherwise return hidden inputs even if pattern doesn't match
              if(t && t.toLowerCase() === 'hidden') return el;
              // otherwise keep as fallback
              return el;
            }catch(e){/* ignore */}
          }
          return null;
        }

        // 1) try inside the widget element
        var tokenEl = findIn(alt);
        // 2) try inside shadowRoot if available
        if(!tokenEl){
          try{ if(alt.shadowRoot) tokenEl = findIn(alt.shadowRoot); }catch(e){}
        }
        // 3) try document-wide (some implementations render inputs outside the custom element)
        if(!tokenEl){
          try{
            var docCandidate = document.querySelector('input[type="hidden"][name="altcha"]') || document.querySelector('input[name="altcha"]') || document.querySelector('input[type="hidden"][name*="altcha"]') || document.querySelector('input[data-altcha]');
            tokenEl = docCandidate;
            if(tokenEl){
              // avoid checkboxes
              var tt = (tokenEl.getAttribute && tokenEl.getAttribute('type')) || tokenEl.type || '';
              if(tt && tt.toLowerCase() === 'checkbox') tokenEl = null;
            }
          }catch(e){}
        }
        // 4) try widget API (if provided)
        if(!tokenEl){
          try{
            if(typeof alt.getConfiguration === 'function'){
              var cfg = alt.getConfiguration();
              if(cfg){
                if(cfg.token) return cfg.token;
                if(cfg.value) return cfg.value;
                if(cfg.challenge) return cfg.challenge;
              }
            }
          }catch(e){}
        }

        if(tokenEl){
          return tokenEl.value || tokenEl.getAttribute('value') || tokenEl.getAttribute('data-value') || null;
        }

        // 5) fallback: attempt to extract from element HTML (last resort)
        try{
          var html = alt.innerHTML || '';
          var m = html.match(/name=["']altcha["'][^>]*value=["']([^"']+)["']/);
          if(m) return m[1];
        }catch(e){}

        return null;
      }

      function bindAlt(alt){
        if(!alt) return;

        alt.addEventListener('statechange', function(ev){
          try{
            if(!(ev && ev.detail && ev.detail.state === 'verified')) return;

            var attempts = 0;
            var maxAttempts = 8;
            var delay = 200;

            var tryReadAndSubmit = function(){
              attempts++;
              var token = readTokenFromWidget(alt);
              if(token){
                // Only use expiry provided by the challenge API (stored on the widget).
                // Do NOT accept expiry parsed from the token itself per requirements.
                var expiresEpoch = null;
                try{
                  if(alt && alt._challengeExpires) expiresEpoch = Math.floor(Number(alt._challengeExpires));
                }catch(e){}

                console.log('Parsed token expiry epoch (from API):', expiresEpoch);

                // Only store `altcha` cookie when the API provided an expiry.
                try{
                  if(expiresEpoch){
                    setCookie('security_check_altcha', token, 7, expiresEpoch);
                  }else{
                    console.warn('No API-provided expires_at; skipping setting security_check_altcha cookie');
                  }
                }catch(e){ console.error('setCookie error', e); }

                ensureAltchaField(token);

                var f = document.getElementById('security-form');
                try{
                  if(typeof f.requestSubmit === 'function'){
                    f.requestSubmit();
                  }else{
                    f.submit();
                  }
                }catch(err){
                  console.error('form submit failed, using fetch fallback', err);
                  submitViaFetchFallback(f);
                }
                return;
              }

              if(attempts < maxAttempts){
                setTimeout(tryReadAndSubmit, delay);
              }else{
                console.warn('Altcha token not found after retries; submitting without token');
                var f2 = document.getElementById('security-form');
                try{ if(typeof f2.requestSubmit === 'function') f2.requestSubmit(); else f2.submit(); }
                catch(e){ submitViaFetchFallback(f2); }
              }
            };

            tryReadAndSubmit();
          }catch(e){ console.error('statechange handler error', e); }
        });
      }

      // Parse expiry timestamp from a token or string. Accepts formats like
      // - embedded query param: "...expires=1772572760"
      // - JSON string containing "expires"
      function parseExpiryFromToken(token){
        if(!token) return null;
        try{
          // try JSON
          if((token.indexOf('{')!==-1 || token.indexOf('"')!==-1)){
            try{
              var j = JSON.parse(token);
              if(j && j.expires) return Math.floor(Number(j.expires));
            }catch(e){}
          }

          // try query param style
          var m = token.match(/expires=(\d{9,})/);
          if(m) return parseInt(m[1],10);

          // try float seconds in token
          var m2 = token.match(/(\d{9,}\.\d+)/);
          if(m2) return Math.floor(Number(m2[1]));
        }catch(e){}
        return null;
      }

      // Set a cookie; `expiresEpoch` can be seconds or milliseconds since epoch.
      function setCookie(name, value, days, expiresEpoch){
        try{
          var parts = [encodeURIComponent(name) + '=' + encodeURIComponent(value)];
          var expiresDate = null;
          if(expiresEpoch){
            var num = Number(expiresEpoch);
            if(!isNaN(num)){
              // if epoch looks like seconds, convert to ms
              if(num < 1e12) num = num * 1000;
              expiresDate = new Date(num);
            }
          }
          if(!expiresDate && typeof days === 'number'){
            expiresDate = new Date(Date.now() + (days*24*60*60*1000));
          }
          if(expiresDate) parts.push('Expires=' + expiresDate.toUTCString());
          parts.push('Path=/');
          // mark Secure if on HTTPS
          if(location && location.protocol && location.protocol.indexOf('https')===0) parts.push('Secure');
          parts.push('SameSite=Lax');
          document.cookie = parts.join('; ');
        }catch(e){ console.error('setCookie error', e); }
      }

      // Read cookie value by name
      function getCookie(name){
        try{
          var re = new RegExp('(?:^|; )' + name.replace(/([.$?*|{}()\[\]\\\/\+^])/g,'\\$1') + '=([^;]*)');
          var m = document.cookie.match(re);
          return m ? decodeURIComponent(m[1]) : null;
        }catch(e){ return null; }
      }

      // If an `altcha` cookie already exists, use it to submit the form immediately
      (function submitFromCookieIfPresent(){
        try{
          var cookieToken = getCookie('security_check_altcha');
          if(!cookieToken) return;
          ensureAltchaField(cookieToken);
          var f = document.getElementById('security-form');
          if(!f) return;
          try{
            if(typeof f.requestSubmit === 'function') f.requestSubmit();
            else f.submit();
          }catch(e){
            console.error('form submit from cookie failed, using fetch fallback', e);
            submitViaFetchFallback(f);
          }
        }catch(e){ /* ignore */ }
      })();

      // Wait for altcha element to be present (widget may upgrade asynchronously)
      function waitForAlt(){
        var alt = document.getElementById('altcha');
        if(alt){
          try{
            if(!alt.customfetch){
              alt.customfetch = async function(url){
                // If an altcha cookie already exists, do not call challenge API.
                try{
                  var cookieToken = getCookie('security_check_altcha');
                  if(cookieToken){
                    var body = JSON.stringify({ from_cookie: true });
                    return new Response(body, { headers: { 'Content-Type': 'application/json' }, status: 200 });
                  }
                }catch(e){ /* ignore cookie read errors and continue */ }

                const challengeUrl = '/v1/challenge';
                try{
                  const resp = await fetch(challengeUrl, { credentials: 'same-origin' });

                  // Only reveal the widget when the challenge fetch succeeded.
                  try{
                    if(resp && resp.ok){ if(alt && alt.style) alt.style.display = ''; }
                  }catch(e){ /* ignore */ }

                  // attempt to read JSON to capture `expires_at` and store it on the widget
                  try{
                    const json = await resp.clone().json();
                    // Only accept explicit `expires_at` from the challenge API.
                    if(json && json.expires_at){
                      var maybe = Math.floor(Number(json.expires_at));
                      if(!isNaN(maybe)) alt._challengeExpires = maybe;
                    }
                  }catch(e){ /* ignore JSON parse failures */ }
                  return resp;
                }catch(err){
                  console.error('widget.customfetch failed, falling back to default fetch for', url, err);
                  try{ return await fetch(url); }catch(e){ return Promise.reject(e); }
                }
              };
            }
          }catch(e){ console.error('set customfetch failed', e); }

          bindAlt(alt);
        }else setTimeout(waitForAlt, 200);
      }

      try{ waitForAlt(); }catch(e){ console.error('failed to bind altcha listener', e); }
    })();
  </script>
