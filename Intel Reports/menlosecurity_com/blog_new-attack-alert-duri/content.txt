// This wrapper waits for jQuery to be fully loaded
$(document).ready(function() {

    // --- Dedicated Accordion Script (Using your classes - KEEP) ---
    const accSettings = {
      speed: 300,
      oneOpen: true,
      offsetAnchor: false,
      offsetFromTop: 180,
      scrollTopDelay: 400,
      classes: {
        accordion: 'ms-faqs-wrapper',
        header: 'ms-faq-question-container-2',
        item: 'ms-faq-card',
        body: 'ms-faq-answer-wrap',
        icon: 'ms-faq-icon',
        active: 'active', // Define active class here
      }
    };

    const prefix = accSettings.classes; // Define prefix here
    const activeClass = prefix.active; // Define activeClass using prefix

    const accordion = (function(){
      const accordionElem = $(`.${prefix.accordion}`);
      if (!accordionElem.length) {
          // console.warn("Accordion element with class '" + prefix.accordion + "' not found."); // Optional logging
          return { init: function() {}, toggle: function() {} };
      }
      // Use event delegation for potentially dynamic accordions and headers
      accordionElem.off('click.accordion', `.${prefix.header}`).on('click.accordion', `.${prefix.header}`, function(e) {
          e.stopImmediatePropagation();
          const $this = $(this); // The clicked header
          const currentItem = $this.closest(`.${prefix.item}`); // Find the parent item
          if (!currentItem.length) {
              console.error("Could not find parent accordion item for clicked header:", $this);
              return;
          }
          accordion.toggle($this, currentItem); // Pass both header and item

          // Scroll-to-top logic
          if(accSettings.offsetAnchor && currentItem.hasClass(activeClass)) { // Check if the *item* is now active
              setTimeout(() => {
                  $('html, body').animate({scrollTop: $this.offset().top - accSettings.offsetFromTop}, accSettings.speed);
              }, accSettings.scrollTopDelay);
          }
      });


      return {
        // Initialize existing accordion states
        init: function(settings) {
          $.extend(accSettings, settings);

          // Correctly handle initial state for oneOpen
          if(settings.oneOpen && $(`.${prefix.item}.${activeClass}`).length > 1) {
            $(`.${prefix.item}.${activeClass}:not(:first)`).removeClass(activeClass).find(`.${prefix.icon}`).removeClass(activeClass);
          }
           $(`.${prefix.item}`).each(function() {
                const item = $(this);
                const body = item.find(`.${prefix.body}`);
                const icon = item.find(`.${prefix.header} .${prefix.icon}`); // Find icon within the header of this item
                if (item.hasClass(activeClass)) {
                    body.show();
                    icon.addClass(activeClass);
                } else {
                    body.hide();
                    icon.removeClass(activeClass);
                }
            });
        },
        // Toggle function now accepts the clicked header ($this) and its parent item (currentItem)
        toggle: function($this, currentItem) {
          const currentBody = currentItem.find(`.${prefix.body}`);
          const currentIcon = $this.find(`.${prefix.icon}`); // Find icon within the clicked header

          if (!currentBody.length) {
              console.error("Could not find accordion body for clicked header:", $this);
              return;
          }

          // Close others only if opening a new one and oneOpen is true
          if(accSettings.oneOpen && !currentItem.hasClass(activeClass)) {
             currentItem.siblings(`.${prefix.item}.${activeClass}`).each(function() {
                const otherItem = $(this);
                otherItem.removeClass(activeClass);
                otherItem.find(`.${prefix.body}`).slideUp(accSettings.speed);
                // Ensure icon within the header is targeted
                otherItem.find(`.${prefix.header} .${prefix.icon}`).removeClass(activeClass);
            });
          }

          // Toggle the current item's state
          currentItem.toggleClass(activeClass);
          currentIcon.toggleClass(activeClass);
          currentBody.stop().slideToggle(accSettings.speed);
        }
      }
    })();

    // Initialize accordion if the function exists
    if (typeof accordion.init === 'function') {
        accordion.init(accSettings);
    }
    // --- End Accordion Script ---


    // --- Sticky Nav Smooth Scroll & Dynamic Offset (KEEP) ---
    const promoBar = document.querySelector('.purple_top-bar-wrapper');
    const mainNav = document.querySelector('.menlo_navbar_01_component');
    const subNav = document.querySelector('.sub-navigation.sticky-sub-nav');
    const subNavLinks = document.querySelectorAll('.sticky-nav-wrapper a[href^="#"]');

    function getStickyNavHeight() {
        const promoBarHeight = (promoBar && window.getComputedStyle(promoBar).display !== 'none') ? promoBar.offsetHeight : 0;
        const mainNavHeight = mainNav ? mainNav.offsetHeight : 0;
        const subNavHeight = subNav ? subNav.offsetHeight : 0;
        return promoBarHeight + mainNavHeight + subNavHeight + 20;
    }
    function getAbsoluteTop(element) {
        if (!element) return 0; // Add check for null element
        let topPosition = 0;
        let currentElement = element;
        while (currentElement) {
            topPosition += currentElement.offsetTop;
            currentElement = currentElement.offsetParent;
        }
        return topPosition;
    }
    if (subNavLinks.length > 0) {
        subNavLinks.forEach(link => {
            link.addEventListener('click', function(e) {
                e.preventDefault();
                e.stopImmediatePropagation();
                const targetId = this.getAttribute('href');
                // Basic check for valid selector
                if (!targetId || targetId === "#" || !targetId.startsWith("#")) return;
                try {
                    const targetElement = document.querySelector(targetId);
                    if (targetElement) {
                        const totalOffset = getStickyNavHeight();
                        let targetPosition = getAbsoluteTop(targetElement);
                        window.scrollTo({
                            top: targetPosition - totalOffset,
                            behavior: 'smooth'
                        });
                    } else {
                        // console.warn("Target element for sticky nav link not found:", targetId); // Optional logging
                    }
                } catch (error) {
                    console.error("Error finding target element for sticky nav:", targetId, error);
                }
            });
        });
    }
    // --- End Sticky Nav ---


    // --- Dynamic Testimonial Schema Generation (KEEP LOGIC) ---
    const testimonialElements = document.querySelectorAll('.f-testimonial-card');
    const quotationsArray = [];

    if (testimonialElements.length > 0) {
        testimonialElements.forEach(item => {
            const quoteText = item.querySelector('.cs-quote')?.textContent.trim();
            const authorOrCompanyName = item.querySelector('.cs_author')?.textContent.trim();
            const roleOrCompanyDescription = item.querySelector('.f-paragraph-role-and-company')?.textContent.trim();

            if (quoteText && authorOrCompanyName) {
                let authorObject = {};
                if (roleOrCompanyDescription) {
                    let potentialTitle = roleOrCompanyDescription.split(',')[0].trim();
                    let potentialCompany = roleOrCompanyDescription.includes(',') ? roleOrCompanyDescription.substring(roleOrCompanyDescription.indexOf(',') + 1).trim() : null;

                    authorObject = { "@type": "Person", "name": authorOrCompanyName, "jobTitle": potentialTitle || undefined };
                    if (potentialCompany) {
                        authorObject.worksFor = { "@type": "Organization", "name": potentialCompany };
                    }
                    else if (["Premier Miton", "Synapxe", "MTR Corporation", "Fortune 100 Financial Services Firm", "Fortune 50 Global Investment Firm", "Top Credit Card Issuer"].includes(authorOrCompanyName)) {
                        authorObject = { "@type": "Organization", "name": authorOrCompanyName, "description": roleOrCompanyDescription || undefined };
                    }
                } else {
                    // Assume Organization if no role/company description
                    authorObject = { "@type": "Organization", "name": authorOrCompanyName };
                }
                quotationsArray.push({ "@type": "Quotation", "text": quoteText, "author": authorObject });
            }
        });
    }
    // No need to log "not found" on every page load unless debugging

    // --- Inject Testimonial Schema ONLY IF Found (FIXED & IMPROVED) ---
    if (quotationsArray.length > 0) {
        const testimonialSchema = {
            "@context": "https://schema.org",
            "@graph": quotationsArray
        };

        // Check if a script added BY THIS LOGIC already exists
        let dynamicSchemaScript = document.querySelector('script[type="application/ld+json"][data-dynamic-schema="testimonial-quotes"]');

        // Check if ANY page-specific schema might exist (more robust check)
         let pageSpecificSchemaExists = Array.from(document.querySelectorAll('script[type="application/ld+json"]')).some(script => {
             // Avoid checking scripts added by this logic itself
             if (script.hasAttribute('data-dynamic-schema') && script.getAttribute('data-dynamic-schema') === 'testimonial-quotes') {
                 return false;
             }
             try {
                 const content = JSON.parse(script.textContent);
                 // Look for common page-specific types OR if it's not just simple Organization/WebSite graph
                 return content && (
                     (content['@type'] && ['WebPage', 'Service', 'Product', 'Article', 'FAQPage', 'HowTo', 'Recipe'].includes(content['@type'])) ||
                     (content['@graph'] && content['@graph'].some(item => item['@type'] && ['WebPage', 'Service', 'Product', 'Article', 'FAQPage', 'HowTo', 'Recipe'].includes(item['@type'])))
                 );
             } catch(e) { return false; } // Ignore scripts with invalid JSON
        });


        // Only inject if NO dynamic quote script exists AND NO other page-specific schema seems present
        if (!dynamicSchemaScript && !pageSpecificSchemaExists) {
            const schemaScript = document.createElement('script');
            schemaScript.type = 'application/ld+json';
            schemaScript.setAttribute('data-dynamic-schema', 'testimonial-quotes'); // More specific attribute
            schemaScript.textContent = JSON.stringify(testimonialSchema);
            // Append to head
            document.head.appendChild(schemaScript);
            console.log(`Dynamic schema: Injected ${quotationsArray.length} quotations into <head>.`);
        } else if (dynamicSchemaScript) {
             // console.log("Dynamic testimonial schema script already exists. Skipping injection."); // Optional logging
        } else if (pageSpecificSchemaExists) {
            // console.log("Page-specific schema detected. Skipping dynamic testimonial injection to avoid conflicts."); // Optional logging
        }
    }
    // --- End Schema ---

});

// Accordion Settings 
const accSettings = {
  speed: 300, // Animation speed
  oneOpen: true, // Close all other accordion items if true
  offsetAnchor: true, // Activate scroll to top for active item
  offsetFromTop: 180, // In pixels â Scroll to top at what distance
  scrollTopDelay: 400, // In Milliseconds â Delay before scroll to top 
  
  classes: {
  	accordion: 'js-accordion',
    header: 'js-accordion-header',
    item: 'js-accordion-item',
    body: 'js-accordion-body',
    icon: 'js-accordion-icon',
    active: 'active',
  }
};


const prefix = accSettings.classes

const accordion = (function(){
  const accordionElem = $(`.${prefix.accordion}`)
  const accordionHeader = accordionElem.find(`.${prefix.header}`)
  const accordionItem = $(`.${prefix.item}`)
  const accordionBody = $(`.${prefix.body}`)
  const accordionIcon = $(`.${prefix.icon}`)
  const activeClass = prefix.active
    
  return {
    // pass configurable object literal
    init: function(settings) {
      accordionHeader.on('click', function() {
        accordion.toggle($(this));
        if(accSettings.offsetAnchor) {
        	setTimeout(() => { 
        		$('html').animate({scrollTop: $(this).offset().top-accSettings.offsetFromTop}, accSettings.speed);
    			}, accSettings.scrollTopDelay);
        }
      });
      
      $.extend(accSettings, settings); 
      
      // ensure only one accordion is active if oneOpen is true
      if(settings.oneOpen && $(`.${prefix.item}.${activeClass}`).length > 1) {
        $(`.${prefix.item}.${activeClass}:not(:first)`).removeClass(activeClass).find(`.${prefix.header} > .${prefix.icon}`).removeClass(activeClass);
      }
      // reveal the active accordion bodies
      $(`.${prefix.item}.${activeClass}`).find(`> .${prefix.body}`).show();
      
    },
    
    toggle: function($this) {
      if(accSettings.oneOpen && $this[0] != $this.closest(accordionElem).find(`> .${prefix.item}.${activeClass} > .${prefix.header}`)[0]) {
        $this.closest(accordionElem).find(`> .${prefix.item}`).removeClass(activeClass).find(accordionBody).slideUp(accSettings.speed);
        $this.closest(accordionElem).find(`> .${prefix.item}`).find(`> .${prefix.header} > .${prefix.icon}`).removeClass(activeClass);
      }
      
      // show/hide the clicked accordion item
      $this.closest(accordionItem).toggleClass(`${activeClass}`).find(`> .${prefix.header} > .${prefix.icon}`).toggleClass(activeClass);
      $this.next().stop().slideToggle(accSettings.speed);
    }
  }
})();

$(document).ready(function(){
  accordion.init(accSettings);
});
