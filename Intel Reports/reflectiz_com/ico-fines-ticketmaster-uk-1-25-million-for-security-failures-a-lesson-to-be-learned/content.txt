            function initHeaderBanner() {
                const $banner = document.querySelector('.s-header__banner');
                const headerBannerSessionKey = 'headerBannerSeen';
                const root = document.documentElement;
                const bannerItemSelector = '.s-header__banner-item';
                let isBannerClosed = false; // For checking if closed
                let timer; // For sliding

                if ( !$banner || ( $banner.classList.contains('u-mobile-hidden') && window.innerWidth < 768 ) ||
                    sessionStorage.getItem(headerBannerSessionKey)) {
                    isBannerClosed = true;
                    initHeaderExtraHeight();

                    return
                };

                // Init
                initBannerSlider(4000); // Init slider
                initBannerClose();

                // On load
                initHeaderExtraHeight(); // Adding css var for extra-height

                // On resize
                window.addEventListener('resize', initHeaderExtraHeight)

                /*=========== Header extra height adding ===========*/

                function initHeaderExtraHeight() {
                    // If banner is hidden on mobile
                    if ( $banner.classList.contains('u-mobile-hidden') && window.innerWidth < 768 ) {
                        isBannerClosed = true;
                    }

                    // If value is in params
                    if (!isBannerClosed) {
                        let $currentSlide = $banner.querySelector(`${bannerItemSelector}.is-active`);

                        if ( $currentSlide ) {
                            $banner.style.minHeight = `${$currentSlide.offsetHeight}px`;
                        }

                        let bannerHeight = $banner.offsetHeight;
                        root.style.setProperty('--header-extra-height', `${bannerHeight}px`);
                    } else {
                        root.style.setProperty('--header-extra-height', `0px`);
                        $banner.style.minHeight = '0px';
                        $banner.style.maxHeight = '0px';
                    }
                }

                /*======== End of Header extra height adding ========*/

                /*=========== Init banner close Event ===========*/

                function initBannerClose() {
                    const $bannerClose = $banner.querySelector('.s-header__banner-close'); // Close button

                    $bannerClose.addEventListener('click', initClose); // Init banner close event

                    function initClose() {
                        isBannerClosed = true; // Set banner closed bool to true
                        sessionStorage.setItem(headerBannerSessionKey, 'yes');
                        initHeaderExtraHeight(); // Clear extra height
                    }
                }

                /*======== End of Init banner close Event ========*/

                /*=========== Banner slider ===========*/

                function initBannerSlider(slideInterval = 5000) {
                    const $bannerItems = $banner.querySelectorAll(bannerItemSelector);

                    if ($bannerItems.length === 1) return;

                    $banner.classList.add('is-slider');
                    document.querySelectorAll(`${bannerItemSelector}`)[0].classList.add('is-active');

                    // Add is-next class to the next slide on init
                    let $currentSlide = $banner.querySelector(`${bannerItemSelector}.is-active`);
                    $currentSlide.nextElementSibling.classList.add('is-next');

                    initHeaderExtraHeight();

                    // Init slider interval
                    timer = setInterval(initSlider, slideInterval);

                    $banner.addEventListener('mouseenter', () => {
                        if ( window.innerWidth > 991 ) {
                            // On hover - clear interval
                            clearInterval(timer);
                        }
                    });

                    $banner.addEventListener('mouseleave', () => {
                        if ( window.innerWidth > 991 ) {
                            // On hover out - set interval
                            timer = setInterval(initSlider, slideInterval);
                        }
                    });

                    //Actually define the initSlider()
                    function initSlider(){
                        if ( isBannerClosed ) return; // If banner is closed - do nothing

                        let current = $banner.querySelector(`${bannerItemSelector}.is-active`);
                        let next = current.nextElementSibling ? current.nextElementSibling : current.parentElement.firstElementChild;
                        let nextNext = next.nextElementSibling ? next.nextElementSibling : next.parentElement.firstElementChild;

                        // Swap slider
                        current.classList.remove('is-active');
                        current.classList.add('is-prev');
                        next.classList.remove('is-next');
                        next.classList.add('is-active');
                        nextNext.classList.remove('is-prev')
                        nextNext.classList.add('is-next');

                        initHeaderExtraHeight();
                    };
                }

                /*======== End of Banner slider ========*/
            }

            initHeaderBanner();
