				context.canvas.height

	function emojiRendersEmptyCenterPoint( context, emoji ) {
		// Cleanup from previous test.
		context.clearRect( 0, 0, context.canvas.width, context.canvas.height );

				return false;

	 * @since 4.2.0

				 * Test for Transgender flag compatibility. Added in Unicode 13.
				 *
				 * To test for support, we try to render it, and compare the rendering to how it would look if

				);

				if ( isIdentical ) {
					return false;
				}

				 * Test for Sark flag compatibility. This is the least supported of the letter locale flags,
				 * so gives us an easy test for full support.
				 *
				 * To test for support, we try to render it, and compare the rendering to how it would look if
				 * the browser doesn't render it correctly ([C] + [Q]).
				 */

					context,
					'\uD83C\uDDE8\uD83C\uDDF6', // as the sequence of two code points
					'\uD83C\uDDE8\u200B\uD83C\uDDF6' // as the two code points separated by a zero-width space
				);

				if ( isIdentical ) {
					return false;
				}

				/*
				 * Test for English flag compatibility. England is a country in the United Kingdom, it
				 * does not have a two letter locale code but rather a five letter sub-division code.
				 *
				 * To test for support, we try to render it, and compare the rendering to how it would look if
				 * the browser doesn't render it correctly (black flag emoji + [G] + [B] + [E] + [N] + [G]).
				 */
				isIdentical = emojiSetsRenderIdentically(
					context,
					// as the flag sequence
					'\uD83C\uDFF4\uDB40\uDC67\uDB40\uDC62\uDB40\uDC65\uDB40\uDC6E\uDB40\uDC67\uDB40\uDC7F',
					// with each code point separated by a zero-width space
					'\uD83C\uDFF4\u200B\uDB40\uDC67\u200B\uDB40\uDC62\u200B\uDB40\uDC65\u200B\uDB40\uDC6E\u200B\uDB40\uDC67\u200B\uDB40\uDC7F'
				);

				return ! isIdentical;
			case 'emoji':
				/*
				 * Does Emoji 16.0 cause the browser to go splat?
				 *
				 * To test for Emoji 16.0 support, try to render a new emoji: Splatter.
				 *
				 * The splatter emoji is a single code point emoji. Testing for browser support

		if (
			typeof WorkerGlobalScope !== 'undefined' &&
			self instanceof WorkerGlobalScope

		} else {

		/*
		 * Chrome on OS X added native emoji rendering in M41. Unfortunately,
		 * it doesn't work when the font is bolder than 500 weight. So, we
		 * check for bold rendering support to avoid invisible emoji in Chrome.
		 */
		context.textBaseline = 'top';
		context.font = '600 32px Arial';

		var supports = {};
		tests.forEach( function ( test ) {
			supports[ test ] = browserSupportsEmoji( context, test, emojiSetsRenderIdentically, emojiRendersEmptyCenterPoint );
		} );
		return supports;

			} catch ( e ) {}
		}

					settings.supports.everything && settings.supports[ test ];

				if ( 'flag' !== test ) {
					settings.supports.everythingExceptFlag =
						settings.supports.everythingExceptFlag &&
						settings.supports[ test ];
				}
			}

			settings.supports.everythingExceptFlag =
				settings.supports.everythingExceptFlag &&
				! settings.supports.flag;

			// Sets DOMReady to false and assigns a ready function to settings.
			settings.DOMReady = false;
			settings.readyCallback = function () {
				settings.DOMReady = true;
			};
		} )
		.then( function () {
			return domReadyPromise;
		} )
		.then( function () {
			// When the browser can not render everything we need to load a polyfill.
			if ( ! settings.supports.everything ) {
				settings.readyCallback();

				var src = settings.source || {};

				if ( src.concatemoji ) {
					addScript( src.concatemoji );
				} else if ( src.wpemoji && src.twemoji ) {
					addScript( src.twemoji );
					addScript( src.wpemoji );
				}
			}
		} );
} )( window, document, window._wpemojiSettings );

							<h2>Anubis Networks is back with new C2 server</h2>
						</div>
						<div class="post-time mb-3">
							<span><i><img src="https://securityaffairs.com/wp-content/themes/security_affairs/images/user-icon.svg" alt="" /></i> <a href="https://securityaffairs.com/author/paganinip">Pierluigi Paganini</a></span>
							<span><i><img src="https://securityaffairs.com/wp-content/themes/security_affairs/images/clock-icon.svg" alt="" /></i> July 11, 2022</span>
						</div>
						<img class="img-fluid mb-4" src="https://i0.wp.com/securityaffairs.com/wp-content/uploads/2022/07/anubis.jpg?fit=690%2C461&ssl=1" alt="" />
						<div style="margin-top: 0px; margin-bottom: 0px;" class="sharethis-inline-share-buttons" ></div>
<h2 class="wp-block-heading">A large-scale phishing campaign leveraging the Anubis Network is targeting Brazil and Portugal since March 2022.</h2>



<p>A large-scale phishing campaign is targeting Internet-end users in Brazil and Portugal since March 2022. Anubis Network is a C2 portal developed to control fake portals and aims to steal credentials to fully access the real systems.</p>



<p>This C2 server is controlled by a group of operators that come from the previous analysis in 2022, the various brands being divided among the operators of the group (in a call center&nbsp;<em>modus operandi</em>).</p>



<p>This campaign is <strong><a href="https://seguranca-informatica.pt/the-evolution-of-phishing-infrastructure-a-large-scale-campaign-called-anubis-that-targets-brazilian-and-portuguese-internet-users/#.Ysq_gHbMIQ9">highlighted by Segurança Informática in 2020</a></strong>, and the high-level diagram of this new campaign can be observed below.<gwmw style="display:none;"></p>



<figure class="wp-block-image"><a href="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/anubis-flow-diagram-.png?ssl=1"><img data-recalc-dims="1" decoding="async" src="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/anubis-flow-diagram-.png?ssl=1" alt="" class="wp-image-10515"/></a></figure>



<p><em><strong>Figure 1:</strong> High-level diagram of the ANUBIS phishing network and its components (2020).</em><gwmw style="display:none;"></p>



<p>In detail, this fresh campaign is composed of three crucial operating components:</p>



<ul class="wp-block-list"><li><strong>the delivery vehicle to propagate the landing page in the wild; usually carried out through smishing (SMS) and phishing (email)</strong></li><li><strong>a malicious landing page hosted on a cloud server, composed of a user interface and layout very similar to the real system</strong></li><li><strong>an operation back-end that allows criminals to manage the details of users who have fallen into the trap.</strong></li></ul>



<p>Figure 2 presents an example of an SMS sent to Internet end-users during the ANUBIS social engineering wave. The image is related to an ongoing campaign in Portugal impersonating a specific organization to steal banking credentials.</p>


<div class="wp-block-image">
<figure class="aligncenter"><a href="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/phone.jpg?ssl=1"><img data-recalc-dims="1" decoding="async" src="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/phone.jpg?ssl=1" alt="" class="wp-image-10516"/></a><figcaption><gwmw style="display:none;"></figcaption></figure></div>


<p><em><strong>Figure 2:</strong> Example of SMS sent during the social engineering wave.</em><gwmw style="display:none;"></p>



<p>SMSs are sent based on a list created by the C2 owner, namely:&nbsp;<strong>1kk-rusha-01.txt</strong>.</p>



<p><strong>Fake domains hosted automatically on Cloudflare CDN</strong><gwmw style="display:none;"></p>



<p>The ANUBIS network phishing campaigns are masked through the Cloudflare CDN. Operators can easily make this configuration through an interface that uses the CloudFlare API for configuring new DNS zones.</p>

<figure class="wp-block-image"><a href="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/email.jpg?ssl=1"><img data-recalc-dims="1" decoding="async" src="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/email.jpg?ssl=1" alt="" class="wp-image-10535"/></a></figure>



<p><em><strong>Figure 12:</strong>&nbsp;Anubis Network email temp feature.</em></p>



<p>The landing pages presented to the victims and specific data can be configured on the Anubis Network administrative portal. The path of the folder and the target brand can be observed on this specific page.</p>



<figure class="wp-block-image"><a href="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/fake.jpg?ssl=1"><img data-recalc-dims="1" decoding="async" src="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/fake.jpg?ssl=1" alt="" class="wp-image-10539"/></a></figure>



<figure class="wp-block-image"><a href="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/brands1.jpg?ssl=1"><img data-recalc-dims="1" decoding="async" src="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/brands1.jpg?ssl=1" alt="" class="wp-image-10538"/></a></figure>



<p><em><strong>Figure 13:</strong>&nbsp;Target organizations of Anubis Network C2 server – Jully 2022.</em></p>



<p>Since the malicious network is made up of many people, a channel on Telegram was created in order to provide technical support to operators in the performance of their duties.</p>


<div class="wp-block-image">
<figure class="aligncenter"><a href="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/tele.jpg?ssl=1"><img data-recalc-dims="1" decoding="async" src="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/tele.jpg?ssl=1" alt="" class="wp-image-10534"/></a></figure></div>


<p><em><strong>Figure 14:&nbsp;</strong>Telegram channel created as a technical support channel.</em></p>



<p><strong>The MySQL database</strong><gwmw style="display:none;"></p>



<p>The heart of the ANUBIS network is a MySQL database. This database is used for data synchronization between all components of the malicious ecosystem and maintains everything up-to-date each second.</p>


<div class="wp-block-image">
<figure class="aligncenter"><a href="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/db.jpg?ssl=1"><img data-recalc-dims="1" decoding="async" src="https://i0.wp.com/seguranca-informatica.pt/wp-content/uploads/2022/07/db.jpg?ssl=1" alt="" class="wp-image-10541"/></a></figure></div>


<p><em><strong>Figure 15:</strong> Database schema of the ANUBIS phishing network.</em></p>
