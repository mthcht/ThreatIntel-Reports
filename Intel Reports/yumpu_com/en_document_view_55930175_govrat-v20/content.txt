<script>
    var documentID;
    var documentURL;

    function isDocumentPrivateModalRequired() {
        if ("1") {
            // Old user, don't show modal
            return false;
        }

        if ($("#yp-edit-document-form").hasClass("d-none")) {
            // No edit form, don't show modal
            return false;
        }

        if ($('#yp-privacy-mode').length && $('#yp-privacy-mode').val() !== "private") {
            // Document is not private, don't show modal
            return false;
        }

        if (!$('#edit_id').val() && !documentID) {
            // No document id, don't show modal
            return false;
        }

        return true;
    }

    function showPrivacyModal(showStep1, document) {
        $("#yp-privacy-modal__step1").toggle(showStep1);
        $("#yp-privacy-modal__step2").toggle(!showStep1);

        $("#dpm__point-1").toggleClass("active", showStep1);
        $("#dpm__point-2").toggleClass("active", !showStep1);

        if (!showStep1 && document) {
            setupDocumentFields(document)
        }
        trackView(showStep1);
        $('#yp-document-privacy-modal').modal('show');
    }

    function trackView(isStep1) {
        if (isStep1) return;
        const isFreeUser = false;
        InternalTracking.track({
            event: 'view',
            custom_key: isFreeUser ? 'published_modal_free' : 'published_modal_other',
        });
    }

    function hidePrivacyModal(update = false) {
        if (update) {
            updateMagazinePrivacy();
        }
        $('#yp-document-privacy-modal').modal('hide');
    }

    function navigateAway(to) {
        window.location = to;
    }

    function setupDocumentFields(document) {
        documentID = document.id;
        documentURL = document.secure_url_with_title;
        $("#dpm__input-link").attr("value", documentURL);
        $("#dpm__view-link").attr("href", documentURL);
        $("#dpm__embed-link").attr("href", documentURL + "#embed");
        setupSocialLinks(document);
    }

    function updateMagazinePrivacy() {
        if (typeof update_magazine !== 'function') {
            return
        }
        $.when(update_magazine('privacy_mode')).done(function (res) {
            var response = $.parseJSON(res);
            if (response.success) {
                availibility_options();
            } else {
                $('#yp-visibility-settings-section').find('.yp-js-change-availability-notifier').removeClass('d-none');
                setTimeout(function () {
                    $('.yp-js-change-availability-notifier').addClass('d-none');
                }, 3200);
            }
        });
    }

    function copyUrlToClipboard() {
        navigator.clipboard.writeText(documentURL);

        var notifier = $('.yp-get-notifier-success').html();
        $(notifier).appendTo('.yp-notification-block').removeClass('d-none');
        $('.yp-notification-block').find('.yp-success-msg').html("Copied successfully!");
        hide_notifier();
    }

    function setupSocialLinks(document) {
        if (!document.settings?.site_social_sharing) {
            $("#dpm__share").css("visibility", "hidden");
            return;
        }
        $("#dpm__share").css("visibility", "visible");
        var links = {
            facebook: `https://www.facebook.com/sharer.php?u=${encodeURIComponent(document.secure_url_with_title)}&t=${encodeURIComponent(document.title + ": " + document.description)}`,
            linkedin: `https://www.linkedin.com/shareArticle?mini=true&url=${encodeURIComponent(document.secure_url_with_title)}&title=${encodeURIComponent(document.title + ": " + document.description)}&source=${encodeURIComponent(document.secure_url_with_title)}`,
            twitter: `https://x.com/intent/tweet?text=${encodeURIComponent(document.title + ", " + "a interesting magazine found on Yumpu.com -" + document.secure_url_with_title + "via @yumpu_com")}&via=yumpu_com&related=youbuy_com`,
            pinterest: `https://pinterest.com/pin/create/button/?url=${encodeURIComponent(document.secure_url_with_title)}&media=${encodeURIComponent(document.secure_pinterest_cover)}&description=${encodeURIComponent(document.pinterest_title + ": " + document.description)}`,
        };
        $("#dpm__socials-facebook").attr("href", links.facebook);
        $("#dpm__socials-linkedin").attr("href", links.linkedin);
        $("#dpm__socials-twitter").attr("href", links.twitter);
        $("#dpm__socials-pinterest").attr("href", links.pinterest);
    }

    $(document).ready(function () {
        $("#dpm__submit").click(function () {
            documentID = documentID ? documentID : $('#edit_id').val();
            if (!documentID) {
                return;
            }
            publish_document(documentID).then((response) => {
                if (response.success) {
                    setupDocumentFields(response.document)
                    $("#yp-privacy-modal__step1").hide();
                    $("#yp-privacy-modal__step2").fadeIn();
                    $("#dpm__point-1").removeClass("active");
                    $("#dpm__point-2").addClass("active");
                }
            });
        })

        $("#dpm__close").click(function () {
            hidePrivacyModal(false);
        })

        $("#dpm__decline").click(function () {
            hidePrivacyModal(true);
        })
    })
